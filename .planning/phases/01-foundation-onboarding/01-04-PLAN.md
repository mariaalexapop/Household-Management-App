```typescript
// Browser client (createBrowserClient) — for auth actions in Client Components
supabase.auth.signUp({ email, password })
supabase.auth.signInWithPassword({ email, password })
supabase.auth.signInWithOAuth({ provider: 'google', options: { redirectTo: '/api/auth/callback' } })
supabase.auth.resetPasswordForEmail(email, { redirectTo: `${appUrl}/auth/update-password` })
supabase.auth.updateUser({ password: newPassword })
supabase.auth.signOut()

// Server client (createServerClient) — for callback route
supabase.auth.exchangeCodeForSession(code)  // PKCE code exchange

// Zod schemas to define (src/lib/validations/auth.ts):
loginSchema: { email: z.string().email(), password: z.string().min(8) }
signupSchema: { email: z.string().email(), password: z.string().min(8), confirmPassword: z.string() }.refine(passwords match)
resetPasswordSchema: { email: z.string().email() }
updatePasswordSchema: { password: z.string().min(8), confirmPassword: z.string() }.refine(passwords match)
```
