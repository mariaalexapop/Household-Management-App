<!-- Realtime subscription pattern (from 01-RESEARCH.md Pattern 3) -->
```typescript
// Connection status type
type ConnectionStatus = 'connected' | 'connecting' | 'disconnected'

// React context
interface RealtimeContextValue {
  status: ConnectionStatus
  // In future phases: subscribe to specific tables without a new connection
}

// Channel subscription callbacks
channel.subscribe((status) => {
  if (status === 'SUBSCRIBED') setStatus('connected')
  if (status === 'CHANNEL_ERROR') setStatus('disconnected')
  if (status === 'TIMED_OUT') setStatus('disconnected')
})

// Heartbeat monitoring for silent disconnects
channel.onHeartbeat?.((heartbeatStatus) => {
  if (heartbeatStatus === 'timeout' || heartbeatStatus === 'disconnected') setStatus('disconnected')
  else if (heartbeatStatus === 'ok') setStatus('connected')
})
```
