```typescript
'use server'

import { z } from 'zod'
import { and, desc, eq } from 'drizzle-orm'
import { db } from '@/lib/db'
import { conversations, messages, householdMembers } from '@/lib/db/schema'
import { createClient } from '@/lib/supabase/server'

export interface ActionResult<T = undefined> {
  success: boolean
  error?: string
  data?: T
}

async function getMemberRow(userId: string) {
  const rows = await db
    .select({ id: householdMembers.id, householdId: householdMembers.householdId })
    .from(householdMembers)
    .where(eq(householdMembers.userId, userId))
    .limit(1)
  return rows[0] ?? null
}
```

**Zod schemas**:
```typescript
const createConversationSchema = z.object({
  title: z.string().min(1).max(200).optional(),
})

const conversationIdSchema = z.object({
  conversationId: z.string().uuid(),
})

const saveUserMessageSchema = z.object({
  conversationId: z.string().uuid(),
  content: z.string().min(1).max(10000),
})

const saveAssistantMessageSchema = z.object({
  conversationId: z.string().uuid(),
  content: z.string().max(20000),
  toolCalls: z
    .array(
      z.object({
        name: z.string(),
        input: z.unknown().optional(),
        output: z.unknown().optional(),
      })
    )
    .optional(),
})
```

**Server actions** — implement these six:

1. **createConversation(data)** — validate, auth-check, resolve household, insert `conversations` row with `createdBy: user.id`, return `{ id }`. Reuse `ActionResult<{ id: string }>`.

2. **listConversations()** — no args. Auth + member check. Return up to 50 most recent conversations in the user's household:
   ```typescript
   .select({ id, title, createdAt, updatedAt })
   .from(conversations)
   .where(eq(conversations.householdId, memberRow.householdId))
   .orderBy(desc(conversations.updatedAt))
   .limit(50)
   ```

3. **loadHistory(data)** — validate `conversationIdSchema`. Auth + member check. Verify the conversation belongs to the user's household (select with AND on `id` + `householdId`). Return messages ordered by `createdAt ASC`:
   ```typescript
   .select({ id, role, content, toolCalls, createdAt })
   .from(messages)
   .where(and(
     eq(messages.conversationId, parsed.data.conversationId),
     eq(messages.householdId, memberRow.householdId),
   ))
   .orderBy(asc(messages.createdAt))
   ```

4. **saveUserMessage(data)** — validate, auth, verify conversation in household, insert message with `role: 'user'`, update `conversations.updatedAt = new Date()`. Return `{ id }`.

5. **saveAssistantMessage(data)** — validate, auth, verify conversation in household, insert with `role: 'assistant'` and serialised `toolCalls` JSONB, update `conversations.updatedAt`. Return `{ id }`.

6. **deleteConversation(data)** — validate `conversationIdSchema`, auth, verify conversation in household, delete row (cascade handles messages).

**Rules applied throughout**:
- Every action starts with `const supabase = await createClient(); const { data: { user } } = await supabase.auth.getUser()` and returns `{ success: false, error: 'Not authenticated' }` when missing.
- Every action calls `getMemberRow` and returns `{ success: false, error: 'No household found' }` on null.
- Zod errors surface as `parsed.error.issues[0]?.message ?? 'Validation failed'`.
- NEVER trust caller-supplied `householdId` — always take it from `memberRow.householdId`.
- Use `revalidatePath` sparingly; conversation data is loaded by the chat panel which controls its own refresh. Skip revalidation here.

**Important for Plan 05-05 integration**: The chat API route will call `saveUserMessage` immediately before calling `streamText`, and `saveAssistantMessage` inside `streamText`'s `onFinish` callback. These server actions work fine from route handlers because they use the same `@/lib/supabase/server` auth helper — the route handler cookie context is inherited.
