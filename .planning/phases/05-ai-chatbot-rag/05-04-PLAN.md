1. **Create `src/lib/inngest/functions/process-document.ts`**:

```typescript
import { inngest } from '@/lib/inngest/client'
import { db } from '@/lib/db'
import { documents, documentChunks, notifications } from '@/lib/db/schema'
import { eq } from 'drizzle-orm'
import { createAdminClient } from '@/lib/supabase/admin'
import { embedBatch } from '@/lib/ai/embed'

const CHUNK_SIZE = 1000
const CHUNK_OVERLAP = 200
const EMBED_BATCH_SIZE = 32

export const processDocument = inngest.createFunction(
  {
    id: 'process-document',
    name: 'Process Document for RAG',
    retries: 3,
    concurrency: { limit: 5 },
  },
  { event: 'documents/uploaded' },
  async ({ event, step }) => {
    const { documentId, householdId, storagePath, uploadedBy } = event.data as {
      documentId: string
      householdId: string
      storagePath: string
      uploadedBy: string
    }

    // 1. Download PDF bytes (retried independently)
    const pdfBase64 = await step.run('download-pdf', async () => {
      const admin = createAdminClient()
      const { data, error } = await admin.storage.from('documents').download(storagePath)
      if (error || !data) throw new Error(`Download failed: ${error?.message ?? 'empty'}`)
      const buf = Buffer.from(await data.arrayBuffer())
      return buf.toString('base64')
    })

    // 2. Extract text with pdf-parse v2. Dynamic import is critical —
    //    top-level import breaks serverless (Pitfall 1 in research).
    const fullText = await step.run('extract-text', async () => {
      // IMPORTANT: load the worker entry before the main module
      await import('pdf-parse/worker')
      const pdfParse = (await import('pdf-parse')).default
      const result = await pdfParse(Buffer.from(pdfBase64, 'base64'))
      return result.text ?? ''
    })

    // 3. Chunk 1000 chars / 200 overlap
    const chunks = await step.run('chunk-text', () =>
      Promise.resolve(chunkText(fullText, CHUNK_SIZE, CHUNK_OVERLAP))
    )

    if (chunks.length === 0) {
      // Still mark ready + notify so the user isn't left waiting
      await step.run('mark-empty-doc-ready', async () => {
        await db
          .update(documents)
          .set({ readyForRag: true })
          .where(eq(documents.id, documentId))
        await db.insert(notifications).values({
          householdId,
          userId: uploadedBy,
          type: 'document_ready',
          entityId: documentId,
          message: 'Document processed (no text found)',
        })
      })
      return { documentId, chunks: 0 }
    }

    // 4. Batch embed + insert. Each batch is its own step for isolated retries.
    const batches = batch(chunks, EMBED_BATCH_SIZE)
    let insertedCount = 0

    for (let i = 0; i < batches.length; i++) {
      const batchChunks = batches[i]
      const startIndex = insertedCount
      await step.run(`embed-batch-${i}`, async () => {
        // Use admin client to bypass cookie auth inside Inngest
        const admin = createAdminClient() as unknown as Parameters<typeof embedBatch>[1]
        const embeddings = await embedBatch(batchChunks, admin)
        await db.insert(documentChunks).values(
          batchChunks.map((content, j) => ({
            householdId,
            documentId,
            chunkIndex: startIndex + j,
            content,
            embedding: embeddings[j],
          }))
        )
      })
      insertedCount += batchChunks.length
    }

    // 5. Mark ready + notify
    await step.run('mark-ready-and-notify', async () => {
      await db
        .update(documents)
        .set({ readyForRag: true })
        .where(eq(documents.id, documentId))

      await db.insert(notifications).values({
        householdId,
        userId: uploadedBy,
        type: 'document_ready',
        entityId: documentId,
        message: 'Your document is ready for AI chat',
      })
    })

    return { documentId, chunks: chunks.length }
  }
)

// ---------------------------------------------------------------------------
// Helpers
// ---------------------------------------------------------------------------

function chunkText(raw: string, size: number, overlap: number): string[] {
  const clean = raw.replace(/\s+/g, ' ').trim()
  if (!clean) return []
  if (clean.length <= size) return [clean]
  const out: string[] = []
  let i = 0
  while (i < clean.length) {
    out.push(clean.slice(i, i + size))
    i += size - overlap
  }
  return out
}

function batch<T>(arr: T[], n: number): T[][] {
  const out: T[][] = []
  for (let i = 0; i < arr.length; i += n) out.push(arr.slice(i, i + n))
  return out
}
```
